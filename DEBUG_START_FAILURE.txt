# أوامر تشخيص فشل التشغيل

# 1. عرض محتوى ملف الـ log لمعرفة الخطأ
cat /tmp/loza-client.log

# 2. محاولة تشغيل يدوي لرؤية الخطأ مباشرة
cd /root/var/www/luzasculture/loza-client-master/loza-client-master
npm run start 2>&1 | head -50

# 3. التحقق من وجود ملفات مهمة
ls -la .env.local 2>/dev/null
ls -la .next/BUILD_ID 2>/dev/null
ls -la node_modules/.bin/next 2>/dev/null

# 4. التحقق من package.json
cat package.json | grep -A 3 '"start"'

# 5. التحقق من صلاحيات الملفات
ls -la .next | head -5
